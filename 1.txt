// how to count one in binary number

uint32_t pop_count(uint32_t a) {
    a = (a & 0x55555555) + ((a & aaaaaaaa) >> 1);
    a = (a & 0x33333333) + ((a & cccccccc) >> 2);
    a = (a & 0x0f0f0f0f) + ((a & f0f0f0f0) >> 4);
    a = (a & 0x00ff00ff) + ((a & ff00ff00) >> 8);
    a = (a & 0x0000ffff) + ((a & ffff0000) >> 16);
}
